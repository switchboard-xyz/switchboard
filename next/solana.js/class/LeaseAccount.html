publicstaticcreate

  • create(program: SwitchboardProgram, params: LeaseInitParams, options?: any): Promise<[LeaseAccount, TransactionSignature]>

  • Creates a LeaseAccount and optionally funds it with wrapped tokens.

    Parameters

    • program: SwitchboardProgram

      The SwitchboardProgram instance.

    • params: LeaseInitParams

      Lease initialization parameters including:

      • aggregatorAccount (required): The AggregatorAccount to be used.
      • queueAccount (required): The QueueAccount to be used.
      • jobAuthorities (optional): Array of PublicKey for job authorities.
      • jobPubkeys (optional): Array of PublicKey for job pubkeys.
      • withdrawAuthority (optional): The PublicKey for the account that has permission to withdraw funds.
    • optionaloptions: any

    Returns Promise<[LeaseAccount, TransactionSignature]>

    A Promise that resolves to a tuple containing the LeaseAccount and the corresponding TransactionObject.

    Basic usage example:

    import { LeaseAccount } from '@switchboard-xyz/solana.js';
    const [leaseAccount, leaseInitSignature] = await LeaseAccount.create(program, {
      queueAccount,
      aggregatorAccount,
      fundAmount: 1,
      funderAuthority: null,
      funderTokenWallet: null,
      disableWrap: false,
      withdrawAuthority: null,
      jobPubkeys: null,
      jobAuthorities: null,
    });
    const lease = await leaseAccount.loadData();

staticcreateInstructions

  • createInstructions(program: SwitchboardProgram, payer: PublicKey, params: LeaseInitParams, options?: TransactionObjectOptions): Promise<[LeaseAccount, TransactionObject]>

  • Creates instructions to initialize a LeaseAccount and optionally funds it with wrapped tokens.

    Parameters

    • program: SwitchboardProgram

      The SwitchboardProgram instance.

    • payer: PublicKey

      The PublicKey of the account that will pay for the transaction fees.

    • params: LeaseInitParams

      Lease initialization parameters including:

      • aggregatorAccount (required): The AggregatorAccount to be used.
      • queueAccount (required): The QueueAccount to be used.
      • jobAuthorities (optional): Array of PublicKey for job authorities.
      • jobPubkeys (optional): Array of PublicKey for job pubkeys.
      • withdrawAuthority (optional): The PublicKey for the account that has permission to withdraw funds.
    • optionaloptions: TransactionObjectOptions

    Returns Promise<[LeaseAccount, TransactionObject]>

    A Promise that resolves to a tuple containing the LeaseAccount and the corresponding TransactionObject.

    Basic usage example:

    import { LeaseAccount } from '@switchboard-xyz/solana.js';
    const [leaseAccount, leaseInitTxn] = await LeaseAccount.createInstructions(program, payer, {
      queueAccount,
      aggregatorAccount,
      fundAmount: 1,
      funderAuthority: null,
      funderTokenWallet: null,
      disableWrap: false,
      withdrawAuthority: null,
      jobPubkeys: null,
      jobAuthorities: null,
    });
    const leaseInitSignature = await program.signAndSend(leaseInitTxn);
    const lease = await leaseAccount.loadData();

publicstaticestimatedLeaseTimeRemaining

  • estimatedLeaseTimeRemaining(oracleRequestBatchSize: number, minUpdateDelaySeconds: number, queueReward: BN, leaseBalance: number): [number, Date]

  • Estimate the time remaining on a given lease

    Parameters

    • oracleRequestBatchSize: number

      the number of oracles to request per openRound call, for a given aggregator.

    • minUpdateDelaySeconds: number

      the number of seconds between openRound calls, for a given aggregator.

    • queueReward: BN

      the number of tokens deducted from an aggregator’s lease for each successful openRound call. This is dependent on the queue an aggregator belongs to.

    • leaseBalance: number

      the current balance in a lease in decimal format.

    Returns [number, Date]

    a tuple containing the number of milliseconds left in a lease and the estimated end date
